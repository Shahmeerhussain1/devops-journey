# All the installation of kind or minikube and creation of cluster

#1
Create namsspace monitoring
kubectl create namespace monitoring

#2
install HELM & run the script from installation document
curl -fsSL -o get_helm.sh https://raw.githubusercontent.com/helm/helm/main/scripts/get-helm-4
chmod 700 get_helm.sh
./get_helm.sh

#3
Open HELM chart repo
https://github.com/prometheus-community/helm-charts

#4
Add the repo link
helm repo add prometheus-community https://prometheus-community.github.io/helm-charts

#5
update the repo
helm repo update

#6
install the prometheus stack from prometheus community.
helm install prometheus-stack prometheus-community/kube-prometheus-stack --namespace monitoring --set prometheus.service.nodePort=30000 --set grafana.set.nodePort=31000 --set grafana.service.type=NodePort --set prometheus.service.type=NodePort 

#7
Take a look at the services , look for prometheus and grafana
kubectl get svc -n monitoring

#8
NOW port forward of promethus 9090
kubectl port-forward svc/prometheus-stack-kube-prom-prometheus 9090:9090 -n monitoring --address=0.0.0.0 

#9 
NOW port forward of grafana 80local to 3000 live
kubectl port-forward svc/prometheus-stack-grafana 3000:80 -n monitoring --address=0.0.0.0 &

#10
Get the password of admin of grafana from secrets
kubectl get secret prometheus-stack-grafana -n monitoring -o jsonpath="{.data.admin-password}" | base64 --decode
